# 虚拟机
echo "deb [arch=amd64] http://download.virtualbox.org/virtualbox/debian $(lsb_release -cs) contrib" | sudo tee -a /etc/apt/sources.list.d/virtualbox.list
wget -q https://www.virtualbox.org/download/oracle_vbox_2016.asc -O- | sudo apt-key add -
wget -q https://www.virtualbox.org/download/oracle_vbox.asc -O- | sudo apt-key add -
up
virtualbox=$(curl http://download.virtualbox.org/virtualbox/LATEST.TXT)
wget https://download.virtualbox.org/virtualbox/$virtualbox/Oracle_VM_VirtualBox_Extension_Pack-$virtualbox.vbox-extpack -O /mnt/Executable/Oracle_VM_VirtualBox_Extension_Pack-$virtualbox.vbox-extpack
sudo apt install bash-completion --reinstall
sudo apt -y install virtualbox-6.1
sudo gpasswd -a ${USER} vboxusers

# XMind
google-chrome-stable https://www.xmind.net/zen/download/linux_deb/ & > /dev/null 2>&1
cd ~/Documents/Software/ ; sudo dpkg -i XMind-for-Linux-amd-64bit-*.deb
sudo chown -R 1000:1000 /opt/XMind
cd /opt/XMind/resources
mkdir -p temp_asar ; npx asar extract app.asar temp_asar/
sed -i 's/{TRIAL:"trial",VALID:"valid",EXPIRED:"expired"}/{TRIAL:"trial",VALID:"trial",EXPIRED:"trial"}/g' "temp_asar/main/main.js" "temp_asar/renderer/common.js"
mv app.asar app.asar.bak ; npx asar pack temp_asar/ app.asar ; rm -rf temp_asar/

# typora
