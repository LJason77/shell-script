up

# XMind
google-chrome-stable https://www.xmind.net/zen/download/linux_deb/ &
cd ~/Documents/Software/ ; sudo dpkg -i XMind-for-Linux-amd-64bit-*.deb
sudo chown -R 1000:1000 /opt/XMind
cd $(find -E /mnt XMind$ /)/resources
# /opt/XMind/resources
sudo mkdir -p temp_asar ; sudo npx asar extract app.asar temp_asar/
sudo sed -i 's/{TRIAL:"trial",VALID:"valid",EXPIRED:"expired"}/{TRIAL:"trial",VALID:"trial",EXPIRED:"trial"}/g' "temp_asar/main/main.js" "temp_asar/renderer/common.js"
sudo mv app.asar app.asar.bak ; sudo npx asar pack temp_asar/ app.asar ; sudo rm -rf temp_asar/
